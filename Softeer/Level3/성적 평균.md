### 풀이

```python
import sys
input = sys.stdin.readline

n,k = tuple(map(int,input().split()))
scores = list(map(int,input().split()))

prefix_sum = [0]
for i in range(n):
    tmp = prefix_sum[-1] + scores[i]
    prefix_sum.append(tmp)

avg = 0
for _ in range(k):
    a,b = tuple(map(int,input().split()))
    a -= 1
    avg = (prefix_sum[b] - prefix_sum[a]) / (b-a)
    print(f'{avg:.2f}')
```
